## Code
```c++
#include<iostream>
#include<vector>
using namespace std;

// Function to merge two sorted arrays into a third sorted array
void merge(int arr1[], int n, int arr2[], int m, int arr3[]) {

    int i = 0, j = 0; // Initialize two pointers for arr1 and arr2
    int k = 0; // Pointer for arr3

    // Traverse both arrays and merge them into arr3
    while( i<n && j<m) {
        // If current element of arr1 is smaller, add it to arr3
        if(arr1[i] < arr2[j]){
            arr3[k++] = arr1[i++]; // Assign arr1[i] to arr3[k] and move both pointers forward
        }
        // Otherwise, add the current element of arr2 to arr3
        else{
            arr3[k++] = arr2[j++]; // Assign arr2[j] to arr3[k] and move both pointers forward
        }
    }

    // If there are remaining elements in arr1, copy them into arr3
    while(i<n) {
        arr3[k++] = arr1[i++]; // Copy remaining elements of arr1
    }

    // If there are remaining elements in arr2, copy them into arr3
    while(j<m) {
        arr3[k++] = arr2[j++]; // Copy remaining elements of arr2
    }
}

// Function to print the elements of the array
void print(int ans[], int n) {
    for(int i=0; i<n; i++) {
        cout<< ans[i] <<" "; // Print each element of the array
    }
    cout << endl;
}

int main() {

    // Two sorted input arrays
    int arr1[5] = {1, 3, 5, 7, 9}; // arr1 with 5 elements
    int arr2[3] = {2, 4, 6};       // arr2 with 3 elements

    // Array to hold the merged result (size is n + m = 5 + 3 = 8)
    int arr3[8] = {0}; 

    // Merge arr1 and arr2 into arr3
    merge(arr1, 5, arr2, 3, arr3);

    // Print the merged array
    print(arr3, 8);

    return 0;
}
```

### Example:
```c++
int arr1[5] = {1, 3, 5, 7, 9};  // Sorted array 1
int arr2[3] = {2, 4, 6};        // Sorted array 2

```

##### Execution:
 -Start by comparing the first elements of arr1 and arr2:
`arr1[0] = 1` and `arr2[0] = 2`. Since `1 < 2`, add `1` to `arr3`.
- Now compare `arr1[1] = 3` and `arr2[0] = 2`. Since `2 < 3`, add `2` to `arr3`.
- Continue this process until one of the arrays is fully traversed.
- Once `arr2 `is fully added, copy the remaining elements from `arr1`.

###### Output:
```c++
1 2 3 4 5 6 7 9
```
